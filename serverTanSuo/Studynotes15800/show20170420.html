代码示例：
/**
		 * 100元买100只鸡--公鸡9元2只--母鸡3元1只--小鸡1元5只--每种至少1只--求出有多少种组合
		 */
		// 总钱数 总只数
		double money = 100;
		int count = 100;
		// 公鸡 母鸡 小鸡的单价
		double moneyGJ = 9.0 / 2;
		double moneyMJ = 3.0;
		double moneyXJ = 1.0 / 5;
		// 组合数
		int sum = 0;
		// i表示公鸡的数量 j表示母鸡的数量 k表示小鸡的数量
		for (int i = 1; i < money / moneyGJ; i++) {
			for (int j = 1; j < money / moneyMJ; j++) {
				for (int k = 1; k < money / moneyXJ; k++) {
					double temp = i * moneyGJ + j * moneyMJ + k * moneyXJ;
					if (i + j + k == count && temp == money) {
						System.out.println(i + " " + j + " " + k);
						sum++;
						break;
					}
					if (i + j + k > count || temp > money)
						break;
				}
			}
		}
		System.out.println(sum);
